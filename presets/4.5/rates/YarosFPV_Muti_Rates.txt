rateprofile 0

# rateprofile 0
set rateprofile_name = CINE
set thr_mid = 50
set thr_expo = 0
set rates_type = ACTUAL
set quickrates_rc_expo = OFF
set roll_rc_rate = 7
set pitch_rc_rate = 7
set yaw_rc_rate = 7
set roll_expo = 10
set pitch_expo = 10
set yaw_expo = 5
set roll_srate = 36
set pitch_srate = 36
set yaw_srate = 36
set throttle_limit_type = OFF
set throttle_limit_percent = 100
set roll_rate_limit = 1998
set pitch_rate_limit = 1998
set yaw_rate_limit = 1998
